demo1

import torch

x = torch.tensor(2., dtype=torch.float, requires_grad=True)
print(x)
# tensor(2., requires_grad=True)

y = 4 * torch.pow(x + 1, 2) + 3
print(y)
# tensor(39., grad_fn=<AddBackward0>)

y.backward()  # = y.backward(torch.tensor(2.))
print(x.grad)
# tensor(24.)

y=4(x+1)2+3y = 4(x+1)^2 + 3y=4(x+1)2+3

dydx=8(x+1)\dfrac{{\rm d}y}{{\rm d}x} = 8(x + 1)dxdy​=8(x+1)；dydx∣x=2=24.0\dfrac{{\rm d}y}{{\rm d}x}\bold{|}_{x=2} = 24.0dxdy​∣x=2​=24.0

demo2

import torch

x = torch.ones(2, 2, requires_grad=True)
print(x)
# tensor([[1., 1.],
#         [1., 1.]], requires_grad=True)

y = 3 * torch.pow(x + 1, 2)
print(y)
# tensor([[12., 12.],
#         [12., 12.]], grad_fn=<MulBackward0>)

out = y.mean()
print(out)
# tensor(12., grad_fn=<MeanBackward0>)

out.backward()  # = out.backward(torch.tensor(1.))
print(x.grad)
# tensor([[3., 3.],
#         [3., 3.]])
